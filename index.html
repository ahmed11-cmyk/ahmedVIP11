async function checkReal(){ const user = document.getElementById('username').value; const plat = document.getElementById('platform').value; if(!user) return alert('دخل اليوزر'); document.getElementById('loading').style.display='block'; document.getElementById('result').style.display='none'; try{ // نستخدم API ينطي صورة + بيانات let url = `https://api.socialcounts.org/${plat}/live/${user}`; const res = await fetch(url); const data = await res.json(); let name = data.name || user; let followers = data.followers || 0; let following = data.following || 0; let posts = data.posts || data.videos || 0; let bio = data.bio || "لا يوجد بايو"; let avatar = data.avatar || `https://ui-avatars.com/api/?name=${user}&background=00ff88&color=000`; // لو API ما انطى صورة نجرب نسحبها if(plat === 'instagram' && !data.avatar){ avatar = `https://www.instagram.com/${user}/?__a=1`; } document.getElementById('loading').style.display='none'; document.getElementById('result').style.display='block'; document.getElementById('result').innerHTML = `

${name}

@${user}

${bio}

المتابعين:${followers.toLocaleString()}
يتابع:${following.toLocaleString()}
المنشورات:${posts.toLocaleString()}
نسبة الامان:${followers>5000?90:60}%
`; }catch(e){ document.getElementById('loading').style.display='none'; alert('الحساب خاص او غير موجود. جرب يوزر عام'); } }
